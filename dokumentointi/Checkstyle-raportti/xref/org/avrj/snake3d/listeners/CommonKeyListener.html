1   package org.avrj.snake3d.listeners;
2   
3   import com.badlogic.gdx.Gdx;
4   import com.badlogic.gdx.Input;
5   import com.badlogic.gdx.InputAdapter;
6   
7   /**
8    * Listens keyboard events in every scene
9    */
10  public class CommonKeyListener extends InputAdapter {
11  
12      /**
13       * Overrides the keyDown method
14       *
15       * @param keycode The keycode of the event
16       * @return returns true if any event is handled
17       */
18      @Override
19      public boolean keyDown(int keycode) {
20          if (keycode == Input.Keys.ENTER) {
21              if (!Gdx.graphics.isFullscreen()) {
22                  Gdx.graphics.setDisplayMode(Gdx.graphics.getDisplayModes()[0]);
23              } else {
24                  Gdx.graphics.setDisplayMode(1280, 720, false);
25              }
26  
27              return true;
28          }
29  
30          return false;
31      }
32  }
