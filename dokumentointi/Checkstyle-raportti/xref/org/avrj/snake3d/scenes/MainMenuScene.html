1   package org.avrj.snake3d.scenes;
2   
3   import com.badlogic.gdx.Gdx;
4   import com.badlogic.gdx.graphics.Color;
5   import com.badlogic.gdx.graphics.GL20;
6   import com.badlogic.gdx.graphics.g2d.BitmapFont;
7   import com.badlogic.gdx.graphics.g2d.freetype.FreeTypeFontGenerator;
8   import com.badlogic.gdx.graphics.g2d.freetype.FreeTypeFontGenerator.FreeTypeFontParameter;
9   import com.badlogic.gdx.scenes.scene2d.Stage;
10  import com.badlogic.gdx.scenes.scene2d.ui.Label;
11  import org.avrj.snake3d.Snake3D;
12  import org.avrj.snake3d.listeners.MainMenuKeyListener;
13  
14  public class MainMenuScene extends Snake3DScene {
15  
16      private boolean isDone = false;
17      private final Stage stage;
18      private BitmapFont titleFont, startGameFont, fullScreenFont, viewScoreBoardFont;
19      private Label titleLabel, startGameLabel, fullScreenLabel, viewScoreBoardLabel;
20      private final StringBuilder stringBuilder;
21      private String startGameText, titleText, fullScreenText, viewScoreBoardText;
22  
23      public MainMenuScene(Snake3D snake3d) {
24          super(snake3d);
25  
26          startGameText = "Press [SPACE] to start";
27          titleText = "Snake3D";
28          fullScreenText = "[ENTER] to enter fullscreen";
29          viewScoreBoardText = "[S] to view scoreboard";
30  
31          stage = new Stage();
32          stringBuilder = new StringBuilder();
33  
34          loadFonts();
35          createLabels();
36  
37          snake3d.getInputMultiplexer().addProcessor(new MainMenuKeyListener(snake3d));
38      }
39  
40      private void createLabels() {
41          titleLabel = new Label(" ", new Label.LabelStyle(titleFont, Color.WHITE));
42          titleLabel.setPosition(stage.getWidth() / 2 - (titleFont.getBounds(titleText).width / 2), stage.getHeight() / 2);
43  
44          startGameLabel = new Label(" ", new Label.LabelStyle(startGameFont, Color.WHITE));
45          startGameLabel.setPosition(stage.getWidth() / 2 - (startGameFont.getBounds(startGameText).width / 2), stage.getHeight() / 2 - 100);
46  
47          fullScreenLabel = new Label(" ", new Label.LabelStyle(fullScreenFont, Color.WHITE));
48          fullScreenLabel.setPosition(stage.getWidth() / 2 - (fullScreenFont.getBounds(fullScreenText).width / 2), 100);
49  
50          viewScoreBoardLabel = new Label(" ", new Label.LabelStyle(viewScoreBoardFont, Color.WHITE));
51          viewScoreBoardLabel.setPosition(stage.getWidth() / 2 - (viewScoreBoardFont.getBounds(viewScoreBoardText).width / 2), stage.getHeight() / 2 - 170);
52  
53          stage.addActor(titleLabel);
54          stage.addActor(startGameLabel);
55          stage.addActor(fullScreenLabel);
56          stage.addActor(viewScoreBoardLabel);
57      }
58  
59      @Override
60      public void dispose() {
61          titleFont.dispose();
62          startGameFont.dispose();
63          fullScreenFont.dispose();
64          viewScoreBoardFont.dispose();
65          stage.dispose();
66      }
67  
68      private void loadFonts() {
69          FreeTypeFontGenerator generator = new FreeTypeFontGenerator(Gdx.files.internal("assets/fonts/AgentOrange.ttf"));
70          FreeTypeFontParameter parameter = new FreeTypeFontParameter();
71  
72          parameter.size = 100;
73          titleFont = generator.generateFont(parameter);
74  
75          parameter.size = 30;
76          startGameFont = generator.generateFont(parameter);
77  
78          parameter.size = 20;
79          fullScreenFont = generator.generateFont(parameter);
80  
81          parameter.size = 20;
82          viewScoreBoardFont = generator.generateFont(parameter);
83  
84          generator.dispose();
85      }
86  
87      @Override
88      public void update(float delta) {
89  
90      }
91  
92      @Override
93      public void draw(float delta) {
94          Gdx.gl.glViewport(0, 0, Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
95          Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT | GL20.GL_DEPTH_BUFFER_BIT);
96  
97          showTitleLabel();
98          showPlayLabel();
99          showToggleFullscreenLabel();
100         showViewScoreBoardLabel();
101 
102         stage.draw();
103     }
104 
105     private void showTitleLabel() {
106         stringBuilder.setLength(0);
107         stringBuilder.append(titleText);
108 
109         titleLabel.setText(stringBuilder);
110     }
111 
112     private void showPlayLabel() {
113         stringBuilder.setLength(0);
114         stringBuilder.append(startGameText);
115 
116         startGameLabel.setText(stringBuilder);
117     }
118 
119     public void showToggleFullscreenLabel() {
120         stringBuilder.setLength(0);
121         stringBuilder.append(fullScreenText);
122 
123         fullScreenLabel.setText(stringBuilder);
124     }
125 
126     public void showViewScoreBoardLabel() {
127         stringBuilder.setLength(0);
128         stringBuilder.append(viewScoreBoardText);
129 
130         viewScoreBoardLabel.setText(stringBuilder);
131     }
132 
133     @Override
134     public boolean isDone() {
135         return isDone;
136     }
137 
138     @Override
139     public void setDone() {
140         this.isDone = true;
141     }
142 
143 }
