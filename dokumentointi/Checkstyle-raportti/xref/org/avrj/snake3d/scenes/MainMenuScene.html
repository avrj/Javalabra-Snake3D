1   package org.avrj.snake3d.scenes;
2   
3   import com.badlogic.gdx.Gdx;
4   import com.badlogic.gdx.graphics.Color;
5   import com.badlogic.gdx.graphics.GL20;
6   import com.badlogic.gdx.graphics.g2d.BitmapFont;
7   import com.badlogic.gdx.graphics.g2d.freetype.FreeTypeFontGenerator.FreeTypeFontParameter;
8   import com.badlogic.gdx.scenes.scene2d.Stage;
9   import com.badlogic.gdx.scenes.scene2d.ui.Label;
10  import org.avrj.snake3d.Snake3D;
11  import org.avrj.snake3d.listeners.MainMenuKeyListener;
12  
13  public class MainMenuScene extends Snake3DScene {
14  
15      private boolean isDone = false;
16      private final Stage stage;
17      private BitmapFont titleFont, startGameFont, fullScreenFont, viewScoreBoardFont;
18      private Label titleLabel, startGameLabel, fullScreenLabel, viewScoreBoardLabel;
19      private final StringBuilder stringBuilder;
20      private String startGameText, titleText, fullScreenText, viewScoreBoardText;
21  
22      public MainMenuScene(Snake3D snake3d) {
23          super(snake3d);
24  
25          stage = new Stage();
26          stringBuilder = new StringBuilder();
27  
28          addLabels();
29  
30          snake3d.getInputMultiplexer().addProcessor(new MainMenuKeyListener(snake3d));
31      }
32  
33      private void setLabelTexts() {
34          startGameText = "Press [SPACE] to start";
35          titleText = "Snake3D";
36          fullScreenText = "[ENTER] to enter fullscreen";
37          viewScoreBoardText = "[S] to view high scores";
38      }
39  
40      private void addLabels() {
41          setLabelTexts();
42          createLabels();
43  
44          stage.addActor(titleLabel);
45          stage.addActor(startGameLabel);
46          stage.addActor(fullScreenLabel);
47          stage.addActor(viewScoreBoardLabel);
48      }
49  
50      private void createLabels() {
51          titleFont = generateFont(100);
52          titleLabel = createLabel(stage.getWidth() / 2 - (titleFont.getBounds(titleText).width / 2), stage.getHeight() / 2, titleFont);
53  
54          startGameFont = generateFont(30);
55          startGameLabel = createLabel(stage.getWidth() / 2 - (startGameFont.getBounds(startGameText).width / 2), stage.getHeight() / 2 - 100, startGameFont);
56  
57          fullScreenFont = generateFont(20);
58          fullScreenLabel = createLabel(stage.getWidth() / 2 - (fullScreenFont.getBounds(fullScreenText).width / 2), 100, fullScreenFont);
59  
60          viewScoreBoardFont = generateFont(20);
61          viewScoreBoardLabel = createLabel(stage.getWidth() / 2 - (viewScoreBoardFont.getBounds(viewScoreBoardText).width / 2), stage.getHeight() / 2 - 170, viewScoreBoardFont);
62      }
63  
64      private Label createLabel(float x, float y, BitmapFont titleFont) {
65          Label label = new Label(" ", new Label.LabelStyle(titleFont, Color.WHITE));
66          label.setPosition(x, y);
67  
68          return label;
69      }
70  
71      @Override
72      public void dispose() {
73          titleFont.dispose();
74          startGameFont.dispose();
75          fullScreenFont.dispose();
76          viewScoreBoardFont.dispose();
77          stage.dispose();
78      }
79  
80      private BitmapFont generateFont(int fontSize) {
81          FreeTypeFontParameter parameters = new FreeTypeFontParameter();
82  
83          parameters.size = fontSize;
84          return snake3d.getFontGenerator().generateFont(parameters);
85      }
86  
87      @Override
88      public void update(float delta) {
89  
90      }
91  
92      @Override
93      public void draw(float delta) {
94          Gdx.gl.glViewport(0, 0, Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
95          Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT | GL20.GL_DEPTH_BUFFER_BIT);
96  
97          setLabels();
98  
99          stage.draw();
100     }
101 
102     private void setLabels() {
103         titleLabel.setText(titleText);
104         startGameLabel.setText(startGameText);
105         fullScreenLabel.setText(fullScreenText);
106         viewScoreBoardLabel.setText(viewScoreBoardText);
107     }
108 
109     @Override
110     public boolean isDone() {
111         return isDone;
112     }
113 
114     @Override
115     public void setDone() {
116         this.isDone = true;
117     }
118 
119 }
