3   import com.badlogic.gdx.math.Vector2;
4   import java.util.ArrayList;
5   import java.util.Random;
6   
7   /**
8    * The apple object (can be "eaten" by the snake)
9    *
10   * @author araiha
11   */
12  public class Apple {
13  
14      private Vector2 applePosition;
15  
16      /**
17       * Sets the position of the apple to (0, 0)
18       */
19      public Apple() {
20          applePosition = new Vector2(0, 0);
21      }
22  
23      public Vector2 getPosition() {
24          return applePosition;
25      }
26  
27      /**
28       * A method to randomize the position of the apple.
29       *
30       * @param goodPositions List of proper positions
31       * @param badPositions List of invalid positions
32       */
33      public void moveAppleToRandomPosition(ArrayList<Vector2> goodPositions, ArrayList<Vector2> badPositions) {
34          Vector2 applePositionOld = applePosition;
35  
36          applePosition = goodPositions.get(new Random().nextInt(goodPositions.size()));
37  
38          while (applePosition == applePositionOld || badPositions.contains(applePosition)) {
39              applePosition = goodPositions.get(new Random().nextInt(goodPositions.size()));
40          }
41      }
42  
43      public void setPosition(Vector2 position) {
44          applePosition = position;
45      }
46  }
