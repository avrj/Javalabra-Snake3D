5   /**
6    * Defines the direction of the snake
7    */
8   public enum SnakeDirection {
9   
10      LEFT(new Vector2(-6f, 0)),
11      RIGHT(new Vector2(6f, 0)),
12      UP(new Vector2(0, -6f)),
13      DOWN(new Vector2(0, 6f));
14  
15      private final Vector2 snakeDirection;
16  
17      private SnakeDirection(Vector2 snakeDirection) {
18          this.snakeDirection = snakeDirection;
19      }
20  
21      public Vector2 getSnakeDirection() {
22          return snakeDirection;
23      }
24  
25      public SnakeDirection getOpposite() {
26          switch (this) {
27              case LEFT:
28                  return RIGHT;
29              case RIGHT:
30                  return LEFT;
31              case UP:
32                  return DOWN;
33              case DOWN:
34                  return UP;
35          }
36  
37          return null;
38      }
39  }
